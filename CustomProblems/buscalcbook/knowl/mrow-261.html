\begin{align*} f'(x) = \amp \lim_{h\rightarrow 0} \dfrac{{\color{blue} f\Big(x+h\Big)} - {\color{red} f\Big(x\Big)} }{h} \\ \amp = \lim_{h\rightarrow 0}\dfrac{ {\color{blue} (x^2 + 2xh + h^2)} - {\color{red} (x^2)} }{h}\\ \amp = \lim_{h\rightarrow 0}\dfrac{ 2xh + h^2 }{h} \\ \amp = \lim_{h\rightarrow 0}\dfrac{ {\color{red} h}(2x + h) }{\color{red} h} \\ \amp = \lim_{h\rightarrow 0} (2x + h) \\ f'(x) \amp = 2x \end{align*}
