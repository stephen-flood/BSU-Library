\begin{equation*} \stackrel{ \scriptstyle\text{to 100} }{ \stackrel{\textstyle\text{S}}{ \scriptstyle\text{from 10} } } C'(x) \approx C(100) - C(10) \end{equation*}

\begin{equation*} \intdx{_{10}^{100} \, C'(x)} = C(100) - C(10) \end{equation*}
